var dailyRate = 49;
   if (groupSize > 5 && groupSize <= 10)
     dailyRate = 49 * 0.9;
   else if (groupSize > 10 && groupSize < 25)
     dailyRate = 49 * 0.8;
   else if (groupSize > 24)
     dailyRate = 49 * 0.75; 
   var groupRate = groupSize * dailyRate;
   document.forms[0].discount.value = groupRate;
